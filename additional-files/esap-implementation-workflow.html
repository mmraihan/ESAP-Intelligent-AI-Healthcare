✓Complete the CITI "Data or Specimens Only Research" training (free online, ~2 hrs)
✓Apply for MIMIC-IV access (Beth Israel / PhysioNet DUA) — tick "BigQuery" option
✓Also apply for: MIMIC-IV-Note, eICU, MIMIC-IV-ECG
✓No sign-up needed for FDA openFDA API (drug adverse events)
⚠ Use your institutional email if you have one — approvals are faster. If not, personal email works fine.

2.1 Build Patient Profile Extractor build ▶

Extract and merge data from MIMIC-IV tables to match ESAP's 10-section patient survey schema.

MIMIC-IV tables used
patients, admissions, diagnoses_icd, prescriptions, labevents, omr, d_icd_diagnoses
Maps to ESAP sections
§1 Demographics · §4 Medical History · §5 Medications · §6 Lab results
def extract_patient_profile(subject_id: int) -> dict: """Build ESAP-compatible profile from MIMIC-IV tables""" # § 1 - Demographics demo_q = f""" SELECT p.subject_id, p.gender, p.anchor_age, a.race, a.language FROM `physionet-data.mimiciv_hosp.patients` p JOIN `physionet-data.mimiciv_hosp.admissions` a ON p.subject_id = a.subject_id WHERE p.subject_id = {subject_id} LIMIT 1 """ # § 4 - Diagnoses (ICD codes → human-readable) diag_q = f""" SELECT d.icd_code, d.icd_version, di.long_title FROM `physionet-data.mimiciv_hosp.diagnoses_icd` d JOIN `physionet-data.mimiciv_hosp.d_icd_diagnoses` di ON d.icd_code = di.icd_code WHERE d.subject_id = {subject_id} ORDER BY d.seq_num """ # § 5 - Medications med_q = f""" SELECT drug, dose_val_rx, dose_unit_rx, route, starttime FROM `physionet-data.mimiciv_hosp.prescriptions` WHERE subject_id = {subject_id} ORDER BY starttime DESC LIMIT 20 """ return { "demographics": client.query(demo_q).to_dataframe(), "diagnoses": client.query(diag_q).to_dataframe(), "medications": client.query(med_q).to_dataframe(), }
2.2 Clinical Notes → Embeddings (RAG setup) ai ▶

Take MIMIC-IV-Note discharge summaries, chunk them, embed with ClinicalBERT, and load into a vector store. This is the core of your RAG pipeline.

→

→

→
Qdrant vector DB
→
RAG retrieval
pip install qdrant-client sentence-transformers langchain # embed_notes.py from sentence_transformers import SentenceTransformer from qdrant_client import QdrantClient, models import pandas as pd model = SentenceTransformer("emilyalsentzer/Bio_ClinicalBERT") qdrant = QdrantClient("localhost", port=6333) # Create collection qdrant.collections.create_collection( collection_name="esap_notes", vectors_config=models.VectorParams(size=768, distance=models.Distance.COSINE) ) def chunk_and_embed_note(subject_id, note_text): # Simple 512-token chunks with 50-token overlap words = note_text.split() chunks = [" ".join(words[i:i+512]) for i in range(0, len(words), 462)] embeddings = model.encode(chunks) qdrant.upsert( collection_name="esap_notes", points=[models.PointStruct( id=hash(f"{subject_id}_{i}") % 2**32, vector=emb.tolist(), payload={"subject_id": subject_id, "chunk": chunk} ) for i, (chunk, emb) in enumerate(zip(chunks, embeddings))] )

3.2 RAG Pipeline — Retrieve + Generate ai ▶

The core AI loop: given a patient ID + current symptoms, retrieve relevant notes from Qdrant, then pass them to the LLM to generate a clinical summary.

import anthropic from qdrant_client import QdrantClient from sentence_transformers import SentenceTransformer embed_model = SentenceTransformer("emilyalsentzer/Bio_ClinicalBERT") qdrant = QdrantClient("localhost", port=6333) llm = anthropic.Anthropic() def esap_rag_query(subject_id: int, current_symptoms: str) -> str: # 1. Embed the query query_vec = embed_model.encode(current_symptoms).tolist() # 2. Retrieve top-5 relevant note chunks for this patient hits = qdrant.search( collection_name="esap_notes", query_vector=query_vec, query_filter=models.Filter( must=[models.FieldCondition( key="subject_id", match=models.MatchValue(value=subject_id) )] ), limit=5 ) context = "\n---\n".join([h.payload["chunk"] for h in hits]) # 3. Generate clinical summary with LLM prompt = f"""You are ESAP, a clinical AI assistant. PATIENT HISTORY CONTEXT (from EHR): {context} CURRENT SYMPTOMS REPORTED TODAY: {current_symptoms} Generate a concise clinical summary for the doctor covering: 1. Key relevant history 2. Possible correlations with current symptoms 3. Flags or alerts (drug interactions, allergy risks) 4. Suggested next steps Be factual, cite the records. Do not hallucinate.""" response = llm.messages.create( model="claude-sonnet-4-6", max_tokens=1000, messages=[{"role": "user", "content": prompt}] ) return response.content[0].text
3.3 Symptom Correlation Agent ai ▶

A separate agent that compares current symptoms against the patient's history and past diagnoses to flag patterns.

def symptom_correlation_agent( current_symptoms: list, diagnosis_history: list, medications: list ) -> dict: prompt = f"""Analyse symptom correlations for a clinical decision support system. Current symptoms: {', '.join(current_symptoms)} Diagnosis history: {', '.join(diagnosis_history)} Current medications: {', '.join(medications)} Return a JSON object with: {{ "correlations": [list of symptom-history matches with confidence 0-1], "drug_symptom_flags": [symptoms possibly caused by current meds], "triage_priority": "low|medium|high|critical", "triage_reason": "one sentence explanation" }} Return ONLY the JSON, no preamble.""" response = llm.messages.create( model="claude-sonnet-4-6", max_tokens=800, messages=[{"role": "user", "content": prompt}] ) import json return json.loads(response.content[0].text)

4.2 Doctor Feedback Capture (the loop) build ▶

This is the most important endpoint — it closes the loop. Every doctor correction is stored as a training signal.

# routers/feedback.py from pydantic import BaseModel from datetime import datetime class FeedbackPayload(BaseModel): patient_id: int doctor_id: str ai_output: str # What the AI said doctor_correction: str # What the doctor changed it to correction_type: str # "diagnosis"|"drug"|"triage"|"summary" accepted: bool # True if doctor accepted AI output notes: str = "" @router.post("/correction") async def capture_feedback(payload: FeedbackPayload): # Store in feedback table record = { **payload.dict(), "timestamp": datetime.utcnow(), "used_for_training": False } db.feedback.insert(record) # Later: batch these for DPO fine-tuning return {"status": "recorded", "id": record["id"]}

5.2 Streaming AI Responses to UI build ▶

Stream the LLM response token-by-token using Server-Sent Events so the doctor sees the summary appearing in real time.

# FastAPI streaming endpoint from fastapi.responses import StreamingResponse @router.post("/ai/summary/stream") async def stream_summary(patient_id: int, symptoms: str): async def generate(): with llm.messages.stream( model="claude-sonnet-4-6", max_tokens=1000, messages=[{"role": "user", "content": build_prompt(patient_id, symptoms)}] ) as stream: for text in stream.text_stream(): yield f"data: {text}\n\n" return StreamingResponse(generate(), media_type="text/event-stream") // React: consume the stream const source = new EventSource(`/ai/summary/stream?patient_id=${id}`); source.onmessage = (e) => setOutput(prev => prev + e.data);

6.2 Activate DPO Fine-tuning (after 100+ feedbacks) ai ▶

Once you have enough doctor corrections, build a DPO dataset and fine-tune a smaller model (Mistral-7B or Llama-3-8B). This proves the closed loop works.

# Build DPO dataset from feedback table def build_dpo_dataset(): corrections = db.query(""" SELECT ai_output, doctor_correction, correction_type FROM feedback WHERE used_for_training = false AND doctor_correction IS NOT NULL ORDER BY timestamp """) dpo_pairs = [] for row in corrections: dpo_pairs.append({ "prompt": row["prompt_used"], "chosen": row["doctor_correction"], # doctor prefers this "rejected": row["ai_output"] # AI originally said this }) # Save as JSONL for TRL DPO trainer with open("dpo_data.jsonl", "w") as f: for pair in dpo_pairs: f.write(json.dumps(pair) + "\n") # Fine-tune with HuggingFace TRL pip install trl transformers datasets # Then: python train_dpo.py --model mistralai/Mistral-7B-v0.1 --data dpo_data.jsonl
